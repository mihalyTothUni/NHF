443template <typename T>
444bool almostEQ(T a, T b) {
445 // eps: ha a relatív, vagy abszolút hiba ettől kisebb, akkor elfogadjuk
446 T eps = 10 * std::numeric_limits<T>::epsilon(); // 10-szer a legkisebb érték
447 if (a == b) return true;
448 if (fabs(a - b) < eps)
449 return true;
450 double aa = fabs(a);
451 double ba = fabs(b);
452 if (aa < ba) {
453 aa = ba;
454 ba = fabs(a);
455 }
456 return (aa - ba) < aa * eps;
457}
